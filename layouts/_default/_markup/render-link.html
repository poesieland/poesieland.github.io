{{- /* #1197 in case of a taxonomy term containing special chars, this will be escaped somewhere inbeetween the term page calling the children shortcode and rendering that Markdown output; so we unescape a sorted characters */}} {{- $path := .Destination }} {{- $htmlchars := dict """ `"` "'" `'` "+" `+` "&" `&` }} {{- range $k, $v := $htmlchars }} {{- $path = replace $path $k $v }} {{- end }} {{- partial "shortcodes/link.html" (dict "page" .Page "url" $path "title" .Title "content" .Text ) }} {{- $u := urls.Parse .Destination -}} {{- /* Check if the link is external (has a protocol or starts with double slash) or internal absolute (starts with simple slash) */ -}} {{- $toIgnore := or (ne $u.Scheme "") (strings.HasPrefix .Destination "/") -}} {{- if not $toIgnore -}} {{- /* Try to fetch the internal page */ -}} {{- $page := .Page.GetPage $u.Path -}} {{- /* If the page does not exist, check if it's a global asset (images, PDFs in /assets) */ -}} {{- if not $page -}} {{- $res := resources.Get $u.Path -}} {{- if not $res -}} {{- /* ⚠️ BROKEN LINK DETECTED: Print a warning to the terminal */ -}} {{- warnf "[BROKEN LINK?] In %s: The link to '%s' does not exist." .Page.File.Path .Destination -}} {{- end -}} {{- end -}} {{- end -}}
